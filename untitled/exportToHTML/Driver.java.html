import java.util.Scanner;

public class Driver {

     private static Scanner input = new Scanner(System.in);
     private Product product = new Product("iphone",);

     public static void main(String[] args) {

         Driver driver = new Driver();
         driver.addProduct();
         driver.printProduct();
     }
     private void addProduct() {
         System.out.print("Enter product Name: ");
         String productName = input.nextLine();

         System.out.print("Enter product Code: ");
         int productCode = input.nextInt();

         System.out.print("Enter product Cost: ");
         double productCost = input.nextDouble();

         System.out.print("Is this product in your current line(Y/N): ");
         char currentLine = input.next().charAt(0);

         boolean inCurrentLine = currentLine == 'Y';
         product = new Product(productName,productCode,productCost,inCurrentLine);
     }
     private void printProduct() {
         System.out.println("------------------");
         System.out.print(product);

     }
}
